CREATE TABLE usuario (
  id VARCHAR(255) PRIMARY KEY,
  name VARCHAR(255),
  email VARCHAR(255),
  password VARCHAR(255)
);

CREATE TABLE telefono (
  id VARCHAR(255) PRIMARY KEY,
  number VARCHAR(255),
  citycode VARCHAR(255),
  countryCode VARCHAR(255),
  usuario_id BIGINT,
  FOREIGN KEY (usuario_id) REFERENCES usuario(id)
);
