CREATE TABLE usuario (
  id BIGINT PRIMARY KEY AUTO_INCREMENT,
  name VARCHAR(255),
  email VARCHAR(255),
  password VARCHAR(255)
);

CREATE TABLE telefono (
  id BIGINT PRIMARY KEY AUTO_INCREMENT,
  number VARCHAR(255),
  citycode VARCHAR(255),
  countryCode VARCHAR(255),
  usuario_id BIGINT,
  FOREIGN KEY (usuario_id) REFERENCES usuario(id)
);
